    package main

import (
	"fmt"
	"sync"
)

func main() {
	var wg sync.WaitGroup
	var counter Counter
	wg.Add(10)
	for i := 0; i < 10; i++ {
      go func() {
          defer wg.Done()
          for j := 0; j < 100000; j++ {
              counter.Incr()
          }
      }()
	}
	wg.Wait()
	fmt.Println(counter.Count())
}

type Counter struct {
	mux   sync.Mutex
	count uint64
}

func (c *Counter) Incr() {
	c.mux.Lock()
	c.count++
	c.mux.Unlock()
}

func (c *Counter) Count() uint64{
	c.mux.Lock()
	defer c.mux.Unlock()
	return c.count
}
